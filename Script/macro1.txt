Sub header()
'
'
    Selection.Find.ClearFormatting
    Selection.Find.Replacement.ClearFormatting
    ActiveWindow.View.ShowFieldCodes = True
    With Selection.Find
        .Text = "MERGEFORMAT[0-9]{1,}"
        .Replacement.Text = "MERGEFORMAT"
        .Forward = True
        .Wrap = wdFindContinue
        .Format = False
        .MatchCase = False
        .MatchWholeWord = False
        .MatchWildcards = True
        .MatchSoundsLike = False
        .MatchAllWordForms = False
    End With
    Selection.Find.Execute Replace:=wdReplaceAll
End Sub

Sub Page_break()
    Dim totalPages As Long
    Dim doc As Document
    Dim rng As Range
    Dim currPage As Long
    Dim nextPage As Long
    Dim pos As Long

    Set doc = ActiveDocument
    
    totalPages = doc.ComputeStatistics(wdStatisticPages)
    If totalPages < 2 Then
        MsgBox "The document has less than 2 pages. Cannot insert section break after second page.", vbExclamation
        Exit Sub
    End If
    
    Set rng = doc.GoTo(What:=wdGoToPage, Which:=wdGoToAbsolute, Count:=2)
    
    currPage = rng.Information(wdActiveEndPageNumber)
    
    pos = rng.Start
    
    Do
        rng.Start = pos
        rng.End = pos + 1
        nextPage = rng.Information(wdActiveEndPageNumber)
        
        If nextPage > currPage Then
            Exit Do

        pos = pos + 1
        If pos > doc.Content.End Then
            Exit Do
        End If
    Loop
    
    Set rng = doc.Range(Start:=pos - 1, End:=pos - 1)
    
    rng.InsertBreak Type:=wdSectionBreakNextPage

End Sub

Sub changefield()
    Dim sec As section
    Dim hdrFooter As HeaderFooter
    Dim para As Paragraph
    Dim rng As Range
    Dim ofPos As Long
    Dim ofRange As Range
    Dim calcField As field
    Dim innerField As field

    For Each sec In ActiveDocument.Sections
        For Each hdrFooter In sec.Footers
            If hdrFooter.Exists Then
                For Each para In hdrFooter.Range.Paragraphs
                    Set rng = para.Range
                    ofPos = InStr(rng.Text, " of ")

                    If ofPos > 0 Then
                        ' Corrected: skip past full " of " (4 chars) + 1 space = 5
                        Set ofRange = rng.Duplicate
                        ofRange.Start = rng.Start + ofPos + 5
                        ofRange.End = rng.End
                        On Error Resume Next
                        ofRange.Delete
                        On Error GoTo 0

                        ' Collapse to insert after " of " and space
                        rng.Start = rng.Start + ofPos + 5
                        rng.End = rng.Start

                        ' Insert outer field: { = ... }
                        Set calcField = ActiveDocument.Fields.Add(Range:=rng, Type:=wdFieldEmpty, Text:="= 1 - 1", PreserveFormatting:=False)
                        calcField.Code.Text = "= "

                        ' Prepare to insert inner field at cursor after '='
                        Dim eqRange As Range
                        Set eqRange = calcField.Code
                        eqRange.Collapse wdCollapseEnd
                        eqRange.Text = " - 10"
                        eqRange.Collapse wdCollapseStart

                        ' Insert inner field: { NUMPAGES \* MERGEFORMAT }
                        Set innerField = ActiveDocument.Fields.Add(Range:=eqRange, Type:=wdFieldEmpty, Text:="NUMPAGES \* MERGEFORMAT", PreserveFormatting:=False)

                        ' Update entire footer
                        hdrFooter.Range.Fields.Update
                        Exit For ' only one match expected per footer
                    End If
                Next para
            End If
        Next hdrFooter
    Next sec

    ' MsgBox "? Inserted: { = { NUMPAGES \* MERGEFORMAT } - 10 } after 'of '."
End Sub

Sub UpdateAllFields()

            shp.TextFrame.TextRange.Fields.Update
        End If
    Next

    ' MsgBox "All fields updated!", vbInformation
End Sub
